## Manual Input Parameters
tension_ref = 3.55  # Reference tension (clear water) Example
tension_measured = 1.55  # Measured tension (turbid water) Example

jtu = (tension_ref - tension_measured) / 0.0165 #This calculation is valid for turbidity values between **0 and 125 JTU**.
k = 0.05 + 0.02 * (jtu ** 0.5)
zeu = 3 / k

print("JTU =", round(jtu, 2))
print("k =", round(k, 4))
print("Zeu =", round(zeu, 2), "m")
